int bianrysearch(int x[], int target, int low, int high)
{
   int m = (low + high)/2;
   while (low <= high)
   {
      m = (low + high)/2;
      if (x[m] > target)
         high = m -1;
      else if (x[m] < target)
         low = m + 1;
      return m;
   }
   return -m;
}
